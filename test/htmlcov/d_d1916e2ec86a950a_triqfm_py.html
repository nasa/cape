27HELP_TRIQFM = r""" 

28``triqfm``: Calculate forces and moments on a ``.triq`` file 

29============================================================= 

30 

31Calculate the integrated forces and moments on a triangulated surface. 

32This can calculate the forces and moments on one or many subcomponents 

33of the annotated surface triangulation and write all the results to 

34parametrically named files. 

35 

36:Usage: 

37 .. code-block:: console 

38  

39 $ triqfm TRIQ COMP1 COMP2 [...] [OPTIONS] 

40 $ triqfm -h 

41 

42:Inputs: 

43 * *TRIQ*: Name of annotated triangulation (TRIQ) file 

44 * *COMP1*: Name or 1-based index of first component to analyze 

45 * *COMP2*: Name or 1-based index of second component to analyze 

46  

47:Examples: 

48  

49 Calculate the forces and moments on the full geometry in 

50 ``"grid.i.triq"``, which is a solution calculated at Mach 0.75. 

51  

52 .. code-block:: console 

53  

54 $ triqfm grid.i.triq --mach 0.75 

55 $ triqfm -m 0.75 

56  

57:Options: 

58 -h, --help 

59 Display this help message and exit 

60  

61 --json JSON 

62 Read options from commented JSON file called *JSON* 

63  

64 --triq TRIQ 

65 Use *TRIQ* as unmapped input file {"grid.i.triq"} 

66  

67 --comps COMPS 

68 List of components to analyze separated by commas 

69  

70 --tri TRI, --map TRI 

71 Use a separate triangulation (Cart3D tri, UH3D, AFLR3 surf, 

72 IDEAS unv, or other TRIQ format) as a map for which triangles to 

73 extract; if used, the component list *COMPS* and config file 

74 *CONFIG* apply to this file 

75  

76 --momentum 

77 Include momentum forces in total 

78  

79 -m MACH, --mach MACH 

80 Use *MACH* as the freestream Mach number {1.0} 

81  

82 --Re REY, --Rey REY 

83 Reynolds number per grid unit {1.0} 

84  

85 --Aref AREF, --RefArea AREF 

86 Reference area for coefficient computation {1.0} 

87  

88 --Lref LREF, --RefLength LREF 

89 Reference length for coefficient computation {1.0} 

90  

91 --bref BREF, --RefSpan BREF 

92 Reference span for coefficient computation {*LREF*} 

93  

94 --xMRP XMRP 

95 Set *x*-coordinate of moment reference point {0.0} 

96  

97 --yMRP YMRP 

98 Set *y*-coordinate of moment reference point {0.0} 

99  

100 --zMRP ZMRP 

101 Set *z*-coordinate of moment reference point {0.0} 

102  

103 --MRP MRP 

104 Moment reference point {"*XMRP*, *YMRP*, *ZMRP*"}  

105  

106:Versions: 

107 * 2017-02-17 ``@ddalle``: Version 1.0 

108"""

111# Main function 

112def triqfm(*a, **kw): 

113 r"""Extract forces and moments from a TRIQ file 

114  

115 Note that both *triq* and *comps* can be overwritten by keyword 

116 arguments even if specified using positional arguments. 

117  

118 :Call: 

119 >>> C = triqfm(triq, *comps, **kw) 

120 :Inputs: 

121 *triq*: {``"grid.i.triq"``} | :class:`str` 

122 Name of TRIQ annotated triangulation file 

123 *comps*: {``None``} | :class:`list` (:class:`str` | :class:`int`) 

124 List of components to analyze (default is to entire surface) 

125 *tri*, *map*: {``None``} | :class:`str` 

126 Separate triangulation file to use for identifying *triq* subset 

127 *c*: {``None``} | :class:`str` 

128 Configuration (XML or JSON format) file to use 

129 *incm*, *momentum*: ``True`` | {``False``} 

130 Include momentum (flow-through) forces in total 

131 *m*, *mach*: {``1.0``} | :class:`float` 

132 Freestream Mach number 

133 *RefArea*, *Aref*: {``1.0``} | :class:`float` 

134 Reference area 

135 *RefLength*, *Lref*: {``1.0``} | :class:`float` 

136 Reference length (longitudinal) 

137 *RefSpan*, *bref*: {*Lref*} | :class:`float` 

138 Reference span (for rolling and yawing moments) 

139 *Re*, *Rey*: {``1.0``} | :class:`float` 

140 Reynolds number per grid unit (units same as *triq.Nodes*) 

141 *gam*, *gamma*: {``1.4``} | :class:`float` > 1 

142 Freestream ratio of specific heats 

143 *xMRP*: {``0.0``} | :class:`float` 

144 *x*-coordinate of moment reference point 

145 *yMRP*: {``0.0``} | :class:`float` 

146 *y*-coordinate of moment reference point 

147 *zMRP*: {``0.0``} | :class:`float` 

148 *z*-coordinate of moment reference point 

149 *MRP* {[*XMRP*, *YMRP*, *ZMRP*]} | :class:`list` 

150 Moment reference point 

151 :Outputs: 

152 *FM*: :class:`dict`\ [:class:`float`] 

153 Dictionary of force and moment coefficients 

154 :Versions: 

155 * 2017-02-16 ``@ddalle``: Version 1.0; :func:`TriqFM` 

156 * 2021-10-14 ``@ddalle``: Version 1.1; in :mod:`cape.triqfm` 

157 """ 

158 # ----------------- 

159 # Sequential Inputs 

160 # ----------------- 

161 # Get TRIQ file name 

162 if len(a) < 1: 

163 # Default 

164 ftriq = "grid.i.triq" 

165 else: 

166 # User-specified 

167 ftriq = a[0] 

168 # Get component list 

169 if len(a) < 2: 

170 # No components 

171 comps = None 

172 else: 

173 # List of components 

174 comps = a[1:] 

175 # -------------- 

176 # JSON Settings 

177 # -------------- 

178 # Check for JSON file 

179 fjson = kw.get("json") 

180 # Read if appropriate 

181 if fjson is None: 

182 # No JSON settings 

183 opts = {} 

184 else: 

185 # Read options from file 

186 opts = loadJSONFile(fjson) 

187 # Load settings from JSON file 

188 ftriq = opts.get("triq", opts.get("TriqFile", ftriq)) 

189 ftri = opts.get("tri", opts.get("TriFile", opts.get("map"))) 

190 fcfg = opts.get("c", opts.get("ConfigFile")) 

191 fo = opts.get("o", opts.get("OutputFile")) 

192 comps = opts.get( 

193 "comps", opts.get("Components", opts.get("CompID", comps))) 

194 incm = opts.get("incm", opts.get("Momentum", False)) 

195 # Freestream conditions 

196 mach = opts.get("m", opts.get("mach", opts.get("Mach", 1.0))) 

197 Rey = opts.get("Re", opts.get("Rey", opts.get("Reynolds", 1.0))) 

198 gam = opts.get("gam", opts.get("gamma", opts.get("Gamma", 1.4))) 

199 # Scale 

200 Aref = opts.get("Aref", opts.get("RefArea", 1.0)) 

201 Lref = opts.get("Lref", opts.get("RefLength", 1.0)) 

202 bref = opts.get("bref", opts.get("RefSpan")) 

203 # Process moment reference point 

204 xMRP = float(opts.get("xMRP", kw.get("xMRP", 0.0))) 

205 yMRP = float(opts.get("yMRP", kw.get("yMRP", 0.0))) 

206 zMRP = float(opts.get("zMRP", kw.get("zMRP", 0.0))) 

207 # Construct total MRP 

208 MRP = opts.get("MRP", [xMRP, yMRP, zMRP]) 

209 # If given as string input, we have to do some work 

210 if "MRP" in kw: 

211 # Split CLI text by comma 

212 MRP = [float(x) for x in kw["MRP"].split(",")] 

213 # -------------- 

214 # Keyword Inputs 

215 # -------------- 

216 # Read settings 

217 ftriq = kw.get("triq", ftriq) 

218 ftri = kw.get("tri", ftri) 

219 ftri = kw.get("map", ftri) 

220 fcfg = kw.get("c", fcfg) 

221 fo = kw.get("o", fo) 

222 incm = kw.get("incm", kw.get("momentum", incm)) 

223 # Check for components from kwargs 

224 kwcomps = kw.get("comps", kw.get("comps", "")).split(",") 

225 if len(kwcomps) > 0: 

226 comps = kwcomps 

227 # Default output file name 

228 if fo is None: 

229 # Strip suffixes 

230 if ftriq.endswith(".i.triq"): 

231 # grid.i.triq -> grid.json 

232 fo = ftriq[:-6] + "json" 

233 elif ftriq.endswith(".triq"): 

234 # grid.triq -> grid.json 

235 fo = ftriq[:-4] + "json" 

236 else: 

237 # grid.uh3d -> grid.uh3d.json 

238 fo = ftriq + ".json" 

239 # Read conditions 

240 mach = kw.get("m", kw.get("mach", mach)) 

241 Rey = kw.get("Re", kw.get("Rey", Rey)) 

242 gam = kw.get("gam", kw.get("gamma", gam)) 

243 # Read scales 

244 Aref = kw.get("Aref", kw.get("RefArea", Aref)) 

245 Lref = kw.get("Lref", kw.get("RefLength", Lref)) 

246 bref = kw.get("bref", kw.get("RefSpan", bref)) 

247 # Fallback reference span 

248 if bref is None: 

249 bref = Lref 

250 # Ensure list of components 

251 if not isinstance(comps, list): 

252 comps = [comps] 

253 # ------ 

254 # Input 

255 # ------ 

256 # Check for a tri file 

257 if ftri is None: 

258 # Read the input TRIQ file 

259 triq = Triq(ftriq, c=fcfg) 

260 # No component map 

261 compmap = {} 

262 else: 

263 # Read the unmapped TRIQ file 

264 triq = Triq(ftriq) 

265 # Read the TRI file 

266 tri = Tri(ftri, c=fcfg) 

267 # Map the component IDs 

268 compmap = triq.MapTriCompID(tri, v=True) 

269 # Initialize output 

270 FM = {} 

271 # ---------- 

272 # Processing 

273 # ---------- 

274 # Set inputs for TriqForces 

275 kwfm = { 

276 "m": float(mach), 

277 "Re": float(Rey), 

278 "gam": float(gam), 

279 "Aref": float(Aref), 

280 "Lref": float(Lref), 

281 "bref": float(bref), 

282 "incm": incm 

283 } 

284 # Loop through components 

285 for comp in comps: 

286 # Process component 

287 if isinstance(comp, (list, np.ndarray)): 

288 # Make up a name 

289 cname = str(comp[0]) 

290 # Translate component numbers if needed 

291 comp = [compmap.get(k, k) for k in comp] 

292 elif (comp is None) or (comp == ""): 

293 # Name of component 

294 cname = "entire" 

295 comp = None 

296 # Which components to process 

297 if ftri is not None: 

298 # Get the list of components from the mapping tri 

299 comp = compmap.values() 

300 else: 

301 # Use the name directly 

302 cname = str(comp) 

303 # If the component is an integer, make sure we use the map 

304 comp = compmap.get(comp, comp) 

305 # Read the forces and moments right from the TRIQ file 

306 FMc = triq.GetTriForces(comp, **kwfm) 

307 # Save it 

308 FM[cname] = FMc 

309 # ------ 

310 # Output 

311 # ------ 

312 # Open the output file 

313 with open(fo, 'w') as fp: 

314 # Dump the results 

315 json.dump(FM, fp, indent=1) 

316 # Output 

317 return FM
