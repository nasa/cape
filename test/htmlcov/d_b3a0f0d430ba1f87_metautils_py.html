24# Local function to perform merger of two dicts 

25# This is a function so that it can be recursed 

26def merge_dict(opts1, opts2): 

27 """Merge two dictionaries, using value from *opts1* in conflict 

28 

29 :Call: 

30 >>> merge_dict(opts1, opts2) 

31 :Inputs: 

32 *opts1*: :class:`dict` 

33 First dictionary 

34 *opts2*: :class:`dict` 

35 Second dictionary 

36 :Versions: 

37 * 2019-04-07 ``@ddalle``: First version 

38 """ 

39 # Loop through options of input dict 

40 for k, v in opts2.items(): 

41 # Check if *k* is in *self* 

42 if k not in opts1: 

43 # New value 

44 opts1[k] = v 

45 continue 

46 # Otherwise check type of input value 

47 if not isinstance(v, dict): 

48 # Save non-dict value 

49 opts1[k] = v 

50 continue 

51 # Get current value 

52 u = opts1[k] 

53 # Then check type of existing value 

54 if not isinstance(u, dict): 

55 # Replace non-dict with dict 

56 opts1[k] = v 

57 continue 

58 # Otherwise, they are both dicts; recurse 

59 merge_dict(u, v)

545 def compare_module(self, mod, modopts): 

546 """Compare specified properties to those of a particular module 

547  

548 :Call: 

549 >>> match = props.compare_module(mod, modopts) 

550 :Inputs: 

551 *props*: :class:`ModulePropDB` 

552 Module property database instance 

553 *mod*: :class:`str` 

554 Name of module to test 

555 *modopts*: :class:`dict` 

556 Dictionary of properties to compare to module *mod* 

557 :Outputs: 

558 *match*: ``True`` | ``False`` 

559 Whether or not *mod* matches all values in *modopts* 

560 :Versions: 

561 * 2019-04-15 ``@ddalle``: First version 

562 """ 

563 # Check input 

564 if not isinstance(modopts, dict): 

565 raise TypeError("Target for comparisons must be a 'dict'") 

566 # Get database for that module 

567 moddb = self.get(mod, {}) 

568 # Get list of available keys 

569 keylist = self.settings.get("Keys", []) 

570 # Loop through input keys 

571 for k, v in modopts.items(): 

572 # Replace "_" with "-" 

573 k = k.replace("_", "-") 

574 # Check if it's a key 

575 if k not in keylist: 

576 raise ValueError( 

577 ("Key '%s' is not a recognized key\n" % k) + 

578 ("Available keys: %s" % keylist)) 

579 # Check comparison 

580 if moddb.get(k) != v: 

581 # Mismatch 

582 return False 

583 # Otherwise, all criteria met 

584 return True

588 """Search for specified properties in a particular module 

589  

590 :Call: 

591 >>> q, keys = props.compare_module_all(mod, *a, **kw)

594 :Inputs:

596 Module property database instance 

597 *k1*: :class:`str` 

598 Name of first key to search (``_`` replaced with ``-``) 

599 *k2*: :class:`str` 

600 Name of second key to search 

601 *v1*: :class:`str` | :class:`any` 

602 Test value 1 (for key *k1* if specified, else any key) 

603 *v2*: :class:`str` | :class:`any` 

604 Test value 2 (for key *k2* if specified) 

605 *kw*: :class:`dict` 

606 Keyword arguments of options to match 

607 :Outputs: 

608 *q*: ``True`` | ``False`` 

609 Whether or not all properties were matched 

610 *keys*: :class:`list`[:class:`str`] 

611 Dictionary of modules that match, with keys that match 

612 :Versions: 

613 * 2019-04-16 ``@ddalle``: First version 

614 """ 

615 # Initialize match flag 

616 q = True 

617 # Initialize match list 

618 keys = [] 

619 # Get list of avialable keys 

620 keylist = self.settings.get("Keys", []) 

621 # Get database for requested module 

622 moddb = self.get(mod, {}) 

623 # Loop through unkeyed search items 

624 for v in a: 

625 # Flag for at least one match of this value 

626 qv = False 

627 # Check if it's a string 

628 v_isstr = typeutils.isstr(v) 

629 # If so, test based on regular expression 

630 if v_isstr: 

631 t = re.compile(v).search 

632 # Loop through entries of module database 

633 for mk, mv in moddb.items(): 

634 # If both are strings, use regular expression search 

635 if v_isstr and typeutils.isstr(mv): 

636 # Regular expression test 

637 qmk = t(mv) is not None 

638 elif (v is None) or (v is True) or (v is False): 

639 # Test using "is" 

640 qmk = (v is mv) 

641 else: 

642 # Test equivalence 

643 qmk = (v == mv) 

644 # If there's a match, update list 

645 if qmk and (mk not in keys): 

646 keys.append(mk) 

647 # Update search flag 

648 qv = qv or qmk 

649 # Update overall match test 

650 q = q and qv 

651 # Loop through keyword search items 

652 for k, v in kw.items(): 

653 # Check if present in module 

654 if k not in moddb: 

655 # No match for at least one key 

656 q = False 

657 continue 

658 # Get test value 

659 mv = moddb[k] 

660 # Check if both values are strings 

661 v_isstr = typeutils.isstr(v) 

662 m_isstr = typeutils.isstr(mv) 

663 # If so, test based on regular expression 

664 if v_isstr and m_isstr: 

665 # Compile regular expression 

666 t = re.compile(v).search 

667 # Perform test 

668 qk = t(mv) is not None 

669 elif (v is None) or (v is True) or (v is False):

673 # Test equivalence
