72 Any value 

73 *types*: :class:`type` | :class:`tuple`\ [:class:`type`] 

74 Type specification as for :func:`isinstance` 

75 :Outputs:

90 r"""Check if a variable is a string (or derivative)

94 :Inputs: 

95 *x*: :class:`any` 

96 Any variable 

97 :Outputs: 

98 *q*: ``True`` | ``False`` 

99 Whether or not *x* is of type :class:`str` or :class:`unicode` or

102 :Versions: 

103 * 2019-03-04 ``@ddalle``: First version

105 # Check version 

106 if PY_MAJOR_VERSION > 2: 

107 # Check just str 

108 return isinstance(x, str)

114# Check for a "list" 

115def isarray(x): 

116 r"""Check if a variable is a list or similar 

117  

118 Accepted types are :class:`list`, :class:`numpy.ndarray`, :class:`tuple`, 

119 or any subclass thereof. 

120  

121 :Call: 

122 >>> q = isarray(x) 

123 :Inputs: 

124 *x*: :class:`any` 

125 Any variable 

126 :Outputs: 

127 *q*: ``True`` | ``False`` 

128 Whether or not *x* is of type :class:`list`, :class:`tuple`, 

129 :class:`np.ndarray`, or any subclass of these three 

130 :Versions: 

131 * 2019-03-04 ``@ddalle``: First version 

132 """ 

133 # Check for traditional list-like things 

134 if isinstance(x, (list, tuple)): 

135 # List or tuple; array-like 

136 return True 

137 elif not isinstance(x, ndarray): 

138 # Not a numeric array 

139 return False 

140 else: 

141 # Check for more than 0 dimensions if NumPy array 

142 return x.ndim > 0

155 :Inputs: 

156 *f*: :class:`any` 

157 Any variable 

158 :Outputs: 

159 *q*: ``True`` | ``False`` 

160 Whether or not *f* is an instance of :class:`file`, 

161 :class:`io.IOBase`, or any subclass 

162 :Versions: 

163 * 2019-12-06 ``@ddalle``: First version 

164 """ 

165 # Check version 

166 if PY_MAJOR_VERSION > 2: 

167 # Check just IOBase 

168 return isinstance(f, io.IOBase) 

169 else: 

170 # Check file as well
