3r""" 

4:mod:`cape.pyfun.plt`: Interface to FUN3D Tecplot PLT files 

5============================================================ 

6 

7This module provides the class :class:`cape.pyfun.plt.Plt`, which  

8intends to read and write Tecplot binary or ASCII PLT files for surface 

9grid solutions from FUN3D. It is based on the generic PLT interface, 

10:class:`cape.bin.plt.Plt`, which does not use the TecIO library to avoid 

11causing unnecessary dependencies for the software. 

12 

13This version of the module has several modifications that are 

14particular to FUN3D solutions. It has a special method for calculating  

15time-averaged *Cp* values from named FUN3D outputs, and it also  

16includes an interface to FUN3D ``.mapbc`` files to interpret names for  

17each boundary zone. 

18 

19This class cannot read any generic ``.plt`` file; it focuses on surface 

20grids with a mix of triangles and quads. In particular it is closely  

21paired with the :mod:`cape.tri` triangulation module. The initial  

22driving cause for creating this module was to read FUN3D boundary  

23solution files and convert them to annotated Cart3D ``triq`` format for 

24input to ``triload`` and other post-processing based on the  

25:mod:`cape.tri` module. 

26 

27:See also: 

28 * :mod:`cape.plt` 

29 * :mod:`cape.tri` 

30 * :mod:`cape.pyfun.mapbc` 

31 * :mod:`pc_Tri2Plt` 

32 * :mod:`pc_Plt2Tri` 

33"""

47# Convert a PLT to TRIQ 

48def Plt2Triq(fplt, ftriq=None, **kw): 

49 r"""Convert Tecplot PLT file to Cart3D annotated tri (TRIQ) 

50  

51 :Call: 

52 >>> Plt2Triq(fplt, ftriq=None, **kw) 

53 :Inputs: 

54 *fplt*: :class:`str` 

55 Name of Tecplot PLT file 

56 *ftriq*: {``None``} | :class:`str` 

57 Name of output file (default: replace extension with  

58 ``.triq``) 

59 *mach*: {``1.0``} | positive :class:`float` 

60 Freestream Mach number for skin friction coeff conversion 

61 *triload*: {``True``} | ``False`` 

62 Whether or not to write a triq tailored for ``triloadCmd`` 

63 *avg*: {``True``} | ``False`` 

64 Use time-averaged states if available 

65 *rms*: ``True`` | {``False``} 

66 Use root-mean-square variation instead of nominal value 

67 :Versions: 

68 * 2016-12-20 ``@ddalle``: First version 

69 """ 

70 # Output file name 

71 if ftriq is None: 

72 # Default: strip .plt and add .triq 

73 ftriq = fplt.rstrip('plt').rstrip('dat') + 'triq' 

74 # TRIQ settings 

75 ll = kw.get('triload', True) 

76 avg = kw.get('avg', True) 

77 rms = kw.get('rms', False) 

78 # Read the PLT file 

79 plt = Plt(fplt) 

80 # Check for mapbc file 

81 fglob = glob.glob("*.mapbc") 

82 # Check for more than one 

83 if len(fglob) > 0: 

84 # Make a crude attempt at sorting 

85 fglob.sort() 

86 # Import the alphabetically last one (should be the same anyway) 

87 kw["mapbc"] = mapbc.MapBC(fglob[0]) 

88 # Attempt to get *cp_tavg* state 

89 if "mach" in kw: 

90 plt.GetCpTAvg(float(kw["mach"])) 

91 # Create the TRIQ interface 

92 triq = plt.CreateTriq(**kw) 

93 # Get output file extension 

94 ext = triq.GetOutputFileType(**kw) 

95 # Write triangulation 

96 triq.Write(ftriq, **kw)

99# Tecplot class 

100class Plt(capeplt.Plt): 

101 r"""Interface for Tecplot PLT files 

102  

103 :Call: 

104 >>> plt = plt.Plt(fname) 

105 :Inputs: 

106 *fname*: :class:`str` 

107 Name of file to read 

108 :Outputs: 

109 *plt*: :class:`cape.pyfun.plt.Plt` 

110 Tecplot PLT interface 

111 *plt.nVar*: :class:`int` 

112 Number of variables 

113 *plt.Vars*: :class:`list`\ [:class:`str`] 

114 List of of variable names 

115 *plt.nZone*: :class:`int` 

116 Number of zones 

117 *plt.Zone*: :class:`int` 

118 Name of each zone 

119 *plt.nPt*: :class:`np.ndarray` (:class:`int`, *nZone*) 

120 Number of points in each zone 

121 *plt.nElem*: :class:`np.ndarray` (:class:`int`, *nZone*) 

122 Number of elements in each zone 

123 *plt.Tris*: :class:`list` (:class:`np.ndarray` (*N*,4)) 

124 List of triangle node indices for each zone 

125 :Versions: 

126 * 2016-11-22 ``@ddalle``: First version 

127 * 2017-03-30 ``@ddalle``: Subclassed to :class:`cape.plt.Plt` 

128 """ 

129 # Calculate cp_tavg 

130 def GetCpTAvg(self, mach, gam=1.4): 

131 r"""Calculate *cp_tavg* if *p_tavg* exists 

132  

133 :Call: 

134 >>> plt.GetCpTAvg(mach, gam=1.4) 

135 :Inputs: 

136 *plt*: :class:`cape.pyfun.plt.Plt` 

137 Tecplot PLT interface 

138 *mach*: :class:`float` 

139 Freestream Mach number 

140 *gam*: {``1.4``} | :class:`float` 

141 Ratio of specific heats 

142 :Versions: 

143 * 2017-05-16 ``@ddalle``: First version 

144 """ 

145 # Check if already present 

146 if 'cp_tavg' in self.Vars: 

147 # Already completed 

148 return 

149 elif 'p_tavg' not in self.Vars: 

150 # No time-averaging information 

151 return 

152 # Append to variables 

153 self.Vars.append('cp_tavg') 

154 self.nVar += 1 

155 # Append column to *qmin* and *qmax* 

156 self.qmin = np.hstack((self.qmin, np.zeros((self.nZone,1)))) 

157 self.qmax = np.hstack((self.qmax, np.zeros((self.nZone,1)))) 

158 # Use time-averaged pressure 

159 k = self.Vars.index('p_tavg') 

160 # Append format for extra column 

161 self.fmt = np.hstack((self.fmt, self.fmt[:,[k]])) 

162 # Loop through states 

163 for n in range(self.nZone): 

164 # Get *cp* 

165 cp = (self.q[n][:,k] - 1/gam)/(0.5*mach*mach) 

166 # Append state 

167 self.q[n] = np.hstack((self.q[n], np.transpose([cp]))) 

168 # Update min/max 

169 self.qmin[n,-1] = np.min(cp) 

170 self.qmax[n,-1] = np.max(cp) 

171 # Update variable locations for this zone if applicable 

172 varlocs = self.VarLocs[n] 

173 # Check if this property has been set 

174 if len(varlocs) <= k: 

175 continue 

176 # Append 

177 self.VarLocs[n] = np.append(varlocs, varlocs[k])
