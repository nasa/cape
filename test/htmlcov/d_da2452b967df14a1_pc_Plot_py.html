19# Function to read the local settings file. 

20def ReadPlotJSON(fname='plot.json'): 

21 """Read plot options from JSON file 

22  

23 :Call: 

24 >>> opts = pc_Plot.ReadPlotJSON(fname='plot.json') 

25 :Inputs: 

26 *fname*: :class:`str` 

27 Name of file to read 

28 :Outputs: 

29 *opts*: :class:`pyCart.options.Plot.Plot 

30 Options interface for iterative history plots 

31 :Versions: 

32 * 2015-02-17 ``@ddalle``: First version 

33 """ 

34 # Check for the file. 

35 if not os.path.isfile(fname): 

36 # Empty settings object. 

37 return Plot() 

38 # Read the file, fail if not present. 

39 f = open(fname) 

40 # Read the settings. 

41 opts = json.load(f) 

42 # Close the file. 

43 f.close() 

44 # Convert to a flowCart object. 

45 return Plot(**opts)

47# Function to process command-line inputs. 

48def ProcessCLI(opts, C=[], **kw): 

49 """Process command-line overrides to options 

50  

51 :Call: 

52 >>> ProcessCLI(opts, C, **kw) 

53 :Inputs: 

54 *opts*: :class:`pyCart.options.Plot.Plot 

55 Options interface for iterative history plots 

56 *C*: :class:`list`\ [:class:`str`] 

57 List of coefficients to plot 

58 *n*: :class:`int` 

59 Number of iterations to plot 

60 *nAve*: :class:`int` 

61 Number of iterations to use for statistics 

62 *nFirst*: :class:`int` 

63  

64 *f*: {``'plot.json'``} | :class:`str` 

65 Name of JSON formatting file to read 

66 *p*: {``'entire'``} | :class:`str` 

67 Name of component to plot  

68 *nRow*: :class:`int` or :class:`str` 

69 Number of rows to use in plot 

70 *nCol*: :class:`int` or :class:`str` 

71 Number of columns to use in plot 

72 *ext*: {``'pdf'``} | ``'pdf'`` | ``'png'`` 

73 File extension  

74 *dpi*: :class:`int` 

75 Dots per inch for output file if raster format is used 

76 *o*: :class:`str` 

77 File name to use (overrides default: ``aero_$comp.$ext`` 

78 *i*: :class:`bool` 

79 Whether or not to display plot interactively 

80 *tag*: :class:`str` 

81 Tag to put in uppler left corner of plot 

82 *dCA*: :class:`float` or :class:`str` or ``None`` 

83 Delta to display on C_A plots 

84 *dCY*: :class:`float` or :class:`str` or ``None`` 

85 Delta to display on C_Y plots 

86 *dCN*: :class:`float` or :class:`str` or ``None`` 

87 Delta to display on C_N plots 

88 *dCLL*: :class:`float` or :class:`str` or ``None`` 

89 Delta to display on C_l plots 

90 *dCLM*: :class:`float` or :class:`str` or ``None`` 

91 Delta to display on C_m plots 

92 *dCLN*: :class:`float` or :class:`str` or ``None`` 

93 Delta to display on C_n plots 

94 *restriction*: :class:`str` 

95 Distribution restriction text to display at bottom of plot 

96 *SBU*: :class:`bool` 

97 Use the restriction text ``'SBU - ITAR'`` 

98 :Versions: 

99 * 2015-02-17 ``@ddalle``: First version 

100 """ 

101 # Coefficient list 

102 if (type(C).__name__ == 'list') and (len(C) > 0): 

103 # Set coefficients. 

104 opts['Coefficients'] = C 

105 # Number of average iterations. 

106 if kw.get('nAve'): opts['nAverage'] = kw['nAve'] 

107 # Number of iterations to plot. 

108 if kw.get('n'): opts['nAverage'] = kw['n'] 

109 # Start and end iterations 

110 if kw.get('nLast'): opts['nLast'] = kw['nLast'] 

111 if kw.get('nFirst'): opts['nFirst'] = kw['nFirst'] 

112 # Check for component plot. 

113 if kw.get('p'): opts.set_PlotComponents([kw['p']]) 

114 # Number of rows and columns. 

115 if kw.get('nRow'): opts['nRow'] = kw['nRow'] 

116 if kw.get('nCol'): opts['nCol'] = kw['nCol'] 

117 # Process restriction flag 

118 if 'SBU' in kw: 

119 # Sensitive but unclassified. 

120 opts['Restriction'] = 'SBU - ITAR' 

121 elif 'ITAR' in kw: 

122 # ITAR 

123 opts['Restriction'] = 'ITAR' 

124 elif 'FOUO' in kw: 

125 # U/FOUO, For Official Use Only 

126 opts['Restriction'] = 'U/FOUO' 

127 elif 'SECRET' in kw: 

128 # Classified, secret 

129 opts['Restriction'] = 'SECRET' 

130 else: 

131 # Use the --restriction option 

132 opts['Restriction'] = kw.get('restriction', '') 

133 # Check for deltas. 

134 if kw.get('dCA'): opts['Deltas']['CA'] = kw['dCA'] 

135 if kw.get('dCY'): opts['Deltas']['CY'] = kw['dCY'] 

136 if kw.get('dCN'): opts['Deltas']['CN'] = kw['dCN'] 

137 if kw.get('dCLL'): opts['Deltas']['CLL'] = kw['dCLL'] 

138 if kw.get('dCLM'): opts['Deltas']['CLM'] = kw['dCLM'] 

139 if kw.get('dCLN'): opts['Deltas']['CLN'] = kw['dCLN']

142def pc_Plot(C=[], **kw): 

143 """Plot a specified list of coefficients or residuals 

144  

145 :Call: 

146 >>> h = pc_Plot(C=[], **kw) 

147 :Inputs: 

148 *C*: :class:`list`\ [:class:`str`] 

149 List of coefficients to plot 

150 *f*: {``'plot.json'``} | :class:`str` 

151 Name of JSON formatting file to read 

152 *p*: {``'entire'``} | :class:`str` 

153 Name of component to plot  

154 *ext*: {``'pdf'``} | ``'pdf'`` | ``'png'`` | ``'svg'`` 

155 File extension  

156 *dpi*: :class:`int` 

157 Dots per inch for output file if raster format is used 

158 *o*: :class:`str` 

159 File name to use (overrides default: ``aero_$comp.$ext`` 

160 *i*: :class:`bool` 

161 Whether or not to display plot interactively 

162 *tag*: :class:`str` 

163 Tag to put in uppler left corner of plot 

164 *restriction*: :class:`str` 

165 Distribution restriction text to display at bottom of plot 

166 *SBU*: :class:`bool` 

167 Use the restriction text ``'SBU - ITAR'`` 

168 :Versions: 

169 * 2014-11-13 ``@ddalle``: First version 

170 """ 

171 # Process file name. 

172 fjson = kw.get('f', 'plot.json') 

173 # Read the options. 

174 opts = ReadPlotJSON(fjson) 

175 # Process the options. 

176 ProcessCLI(opts, C, **kw) 

177 # Get the components. 

178 comps = opts.get_PlotComponents() 

179 

180 # Name of this folder. 

181 fpwd = os.path.split(os.getcwd())[-1] 

182 # Get extension. 

183 ext = kw.get('ext', 'pdf') 

184 # Get file name. 

185 f0 = kw.get('o', 'aero') 

186 

187 # Check for 'case.json' flag. 

188 if os.path.isfile('case.json'): 

189 fc = ReadCaseJSON() 

190 

191 # Read the data. 

192 AP = Aero(comps) 

193 

194 # Loop through components. 

195 for comp in comps: 

196 # New figure. 

197 plt.figure() 

198 # Default tag 

199 ftag = '%s\nComponent=%s' % (fpwd, comp) 

200 

201 # Coefficients to plot. 

202 coeffs = opts.get_PlotCoeffs(comp) 

203 

204 # Initialize dictionary of plot deltas. 

205 d = {} 

206 # Loop through coefficients. 

207 for coeff in coeffs: 

208 d[coeff] = opts.get_PlotDelta(coeff, comp) 

209 

210 # Plot keyword arguments 

211 kwp = { 

212 'nRow': opts.get_nPlotRows(comp), 

213 'nCol': opts.get_nPlotCols(comp), 

214 'n': opts.get_nPlotIter(comp), 

215 'nFirst': opts.get_nPlotFirst(comp), 

216 'nLast': opts.get_nPlotLast(comp), 

217 'nAvg': opts.get_nAverage(comp), 

218 'tag': kw.get('tag', ftag), 

219 'restriction': opts.get_PlotRestriction(comp), 

220 'd': d 

221 } 

222 

223 # Plot the components. 

224 h = AP.Plot(comp, coeffs, **kwp) 

225 

226 # Check the pass flag. 

227 if kw.get('PASS'): h['pass'].set_text('PASS') 

228 # Check for ability to put iteration number. 

229 if os.path.isfile('case.json'): 

230 # Set the iteration flag. 

231 h['iter'].set_text('%i/%i' % 

232 (AP[comp].i[-1], fc.get_LastIter())) 

233 

234 # Output file name. 

235 fname = '%s_%s.%s' % (f0, comp, ext) 

236 

237 # Save the file. 

238 if ext in ['png', 'jpg']: 

239 # Get the dots per inch setting. 

240 idpi = int(kw.get('dpi', 150)) 

241 # Save the figure. 

242 h['fig'].savefig(fname, dpi=idpi) 

243 else: 

244 # Save the figure to vector format. 

245 h['fig'].savefig(fname) 

246 # Check for interactive mode. 

247 if kw.get('i'): 

248 plt.show()
